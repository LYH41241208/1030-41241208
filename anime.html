Anime.js 官網

範例程式碼

<div id="anime-box" style="width:100px;height:100px;background:coral;margin:auto"></div>
<script>
    anime({
        targets: '#anime-box',
        translateX: 250,
        rotate: '1turn',
        duration: 2000,
        easing: 'easeInOutSine'
    });
</script>
