Chart.js 圖表

Chart.js 官網

範例程式碼

<canvas id="myChart"></canvas>
<script>
    const ctx = document.getElementById('myChart').getContext('2d');
    new Chart(ctx, {
        type: 'bar',
        data: {
            labels: ['Red', 'Blue', 'Yellow', 'Green', 'Purple', 'Orange'],
            datasets: [{
                label: '# of Votes',
                data: [12, 19, 3, 5, 2, 3],
                backgroundColor: 'rgba(180, 180, 180, 0.2)',
                borderColor: 'rgba(30, 180, 190, 1)',
                borderWidth: 2
            }]
        }
    });
</script>
